Usage:
```
    <repositories>
        <repository>
            <id>jialtang-github</id>
            <name>The Maven Repository on Github</name>
            <url>https://raw.github.com/jialtang/java_common/maven-repo</url>
        </repository>
    </repositories>

    <dependencies>
        <dependency>
            <groupId>com.jialtang</groupId>
            <artifactId>java_common</artifactId>
            <version>1.0-SNAPSHOT</version>
        </dependency>
    </dependencies>
```
